fun isEven(int n) bool {
    if (n == 0) {
        return true
    }
    return isOdd(n - 1)
}

fun isOdd(int n) bool {
    if (n == 0) {
        return false
    }
    return isEven(n - 1)
}

fun evenOrOdd(int n) void {
    if (isEven(n)) {
        print("n is Even ")
    } else {
        print("n is Odd ")
    }
    print(n)
}

fun loopLimit(int n) int {
    if (n == 0) {
        print("Terminé")
        return 0
    } elif (n % 2 == 0) {
        print("Pair :" + n)
    } else {
        print("Impair :" + n)
    }
    evenOrOdd(n)
    loopLimit(n - 1)
    return 0
}
print(loopLimit(6))
